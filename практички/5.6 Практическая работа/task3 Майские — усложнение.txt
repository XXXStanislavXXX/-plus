int hradHollidays() {
  std::cout << "Введите номер дня недели, с которого начинается месяц: ";
  int startDay;
  std::cin >> startDay;
  if (startDay < 1 || startDay > 7) {
    std::cout << "Номер дня недели должен быть в диапазоне от 1 до 7";
    std::cout << "Введите номер дня недели, с которого начинается месяц: ";
    std::cin >> startDay;
  }
  std::cout << "Введите день: ";
  int day;
  std::cin >> day;
  if (day < 1 || day > 31) {
    std::cout << "День должен быть в диапазоне от 1 до 31";
    std::cout << "ВВедите день: ";
    std::cin >> day;
  }
  if ((startDay - 1 + day) % 7 == 6 || (startDay - 1 + day) % 7 == 0 || (day >= 1 && day <= 5) || (day >= 8 && day <= 10) ) {
      std::cout << "Выходной";
    }
  else {
    std::cout << "Рабочий день";
  }
  return 0;
}
